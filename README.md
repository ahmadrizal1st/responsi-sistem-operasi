# MENJELASKAN MASING-MASING MATERI DARI MODUL

# Bab 1 Perintah Dasar Linux
- whoami : Menampilkan nama user yang sedang login

- date : Menampilkan tanggal dan waktu sistem

- hostname : Menampilkan nama komputer/server

# Bab 2 Sistem File
- pwd : Menampilkan lokasi direktori aktif

- ls : Menampilkan daftar file dan folder

- ls -a : Menampilkan semua file termasuk file tersembunyi 

# Bab 3 Konfigurasi Linux 
- uname -a : Menampilkan informasi kernel dan sistem operasi

- cat /etc/os-release : Menampilkan detail distribusi Linux

- echo $SHELL : Menampilkan shell yang digunakan user 

# Bab 4 Ultilitas Linux
- ar --version : Menampilkan versi utilitas arsip tar

- gzip --version : Menampilkan versi kompresi gzip

- zip -v : Menampilkan versi utilitas zip

# Bab 5 Proses Input Output
- echo "contoh teks" > io.txt : Mengarahkan output ke file (redirect output)

- cat io.txt : Menampilkan isi file

- wc io.txt : Menghitung baris, kata, dan karakter

# Bab 6 Manajemen Proses
- ps : Menampilkan proses yang sedang berjalan

- ps -u : Menampilkan proses milik user

- uptime : Menampilkan lama sistem berjalan & load average
