MODUL 1
whoami = Menampilkan nama user yang sedang login pada sistem.
date = Menampilkan tanggal dan waktu sistem saat ini.

MODUL2
mkdir = Membuat direktori/folder baru dengan nama finda.
ls -l = Menampilkan/melihat daftar file dan direktori secara lengkap.

MODUL 3
cp /etc/finda /etc/finda.backup = Menyalin file konfigurasi finda menjadi file cadangan (backup).
nano /etc/finda = Membuka file finda menggunakan text editor nano untuk diedit.

MODUL 4
gzip -c finda > finda.gz = Mengompres file finda menjadi finda.gz tanpa menghapus file aslinya.
tar -cf arsip.tar finda1 finda2 = Mengarsipkan beberapa file (finda1 dan finda2) ke dalam satu file arsip.tar.

MODUL 5
echo hello > output = Menulis teks hello ke file output dan menimpa isi lama jika file sudah ada.
echo hello >> output = Menambahkan teks hello ke akhir file output tanpa menghapus isi sebelumnya.

MODUL 6
ps -u = Menampilkan daftar proses yang sedang berjalan dengan informasi user.
pstree = Menampilkan hubungan proses parent dan child dalam bentuk pohon (tree).

MODUL 7
history = Menampilkan riwayat perintah yang pernah dijalankan oleh user.
cd /usr/sbin = Untuk melihat prompt usr/sbin
